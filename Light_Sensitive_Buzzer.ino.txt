// Light Sensitive Buzzer using LDR and Arduino

const int ldrPin = A0;       // LDR connected to analog pin A0
const int buzzerPin = 9;     // Buzzer connected to digital pin 9

void setup() {
  pinMode(buzzerPin, OUTPUT);
  Serial.begin(9600);
  Serial.println("Light Sensor Buzzer Initialized");
}

void loop() {
  int ldrValue = analogRead(ldrPin);
  Serial.print("LDR Value: ");
  Serial.println(ldrValue);

  if (ldrValue < 400) {  // Adjust threshold based on your lighting
    digitalWrite(buzzerPin, HIGH);  // Dark environment → buzzer ON
    Serial.println("Dark detected → Buzzer ON");
  } else {
    digitalWrite(buzzerPin, LOW);   // Bright environment → buzzer OFF
    Serial.println("Bright → Buzzer OFF");
  }

  delay(1000);
}
